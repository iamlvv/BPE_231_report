\textbf{Use-case name}     & Leave workspace                          \\ \hline
\textbf{Actor}             & Shared member (can view, share, edit)    \\ \hline
\textbf{Description}       & Rời khỏi workspace                       \\ \hline
\textbf{Trigger}           & Không                                    \\ \hline
\textbf{Pre-conditions} &
  Người dùng đã đăng ký vào hệ thống và đăng nhập thành công, hiện tại đang đứng ở giao diện workspace đã chọn \\ \hline
\textbf{Post-conditions}   & Người dùng thành công rời khỏi workspace \\ \hline
\textbf{Normal flow} &
  \begin{enumerate}
      \item Người dùng ở giao diện của workspace đã chọn
      \item Người dùng chọn nút “Leave workspace”, hệ thống hiển thị modal xác nhận
      \item Người dùng chọn “Yes” để rời khỏi workspace
  \end{enumerate}
  \\ \hline
\textbf{Alternative flows} & 3a. Người dùng chọn “No”                 \\ \hline
\textbf{Exceptions}        & Không                                    \\ \hline
