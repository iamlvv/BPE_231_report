\subsubsection{Custom workspace appearance}

\begin{table}[H]
\centering
\def\arraystretch{2}%
\resizebox{\textwidth}{!}{%
\begin{tabular}{|p{3cm}|p{11cm}|}
\hline
\textbf{Use-case name} &
  Custom workspace appearance \\ \hline
\textbf{Actor} &
  Workspace owner \\ \hline
\textbf{Description} &
  Workspace owner tùy chỉnh giao diện Workspace \\ \hline
\textbf{Trigger} &
  Không \\ \hline
\textbf{Pre-conditions} &
  Workspace owner đã đăng nhập vào hệ thống \\ \hline
\textbf{Post-conditions} &
  Tùy chỉnh giao diện Workspace thành công \\ \hline
\textbf{Normal Flow} &
  \begin{enumerate}
      \item Workspace owner chọn workspace.
      \item Hệ thống hiển thị trang workspace.
      \item Người dùng chọn nút “Workspace Management”
      \item Hệ thống chuyển sang trang Workspace Management
      \item Người dùng chọn nút “Workspace Customization” ở sidebar
      \item Hệ thống hiển thị 2 lựa chọn: Background và Icon
      \item Người dùng chọn Background
      \item Hệ thống hiển thị cửa sổ để người dùng chọn file ảnh cần upload
      \item Người dùng chọn ảnh upload
      \item Hệ thống load ảnh thành công và cài đặt background của workspace 
  \end{enumerate}
  \\ \hline
\textbf{Alternative Flow} &
  \begin{tabular}[c]{@{}l@{}}7.1. Người dùng chọn Icon\\ Bắt đầu tiếp từ bước 8.\\ 10. Hệ thống load ảnh thành công và cài đặt icon của workspace.\end{tabular} \\ \hline
\textbf{Exceptions} &
  Không \\ \hline
\end{tabular}%
}
\end{table}
