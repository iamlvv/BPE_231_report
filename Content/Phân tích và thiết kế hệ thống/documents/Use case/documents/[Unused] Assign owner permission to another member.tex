\subsubsection{Assign owner permission to another member}

\begin{table}[H]
\def\arraystretch{2}%
\centering
\resizebox{\textwidth}{!}{%
\begin{tabular}{|p{3cm}|p{11cm}|}
\hline
\textbf{Use-case name}    & Assign owner permission to another member                                                                               \\ \hline
\textbf{Actor}            & Workspace owner                                                                                                         \\ \hline
\textbf{Description}      & Workspace owner chuyển quyền sở hữu Workspace cho thành viên khác trong Workspace.                                      \\ \hline
\textbf{Trigger}          & Không                                                                                                                   \\ \hline
\textbf{Pre-conditions}   & Workspace owner đã đăng nhập vào hệ thống và trong workspace có ít nhất 1 thành viên khác ngoài owner.                  \\ \hline
\textbf{Post-conditions}  & Chuyển quyền thành công.                                                                                                \\ \hline
\textbf{Normal Flow} &
  \begin{enumerate}
      \item Workspace owner chọn workspace.
      \item Hệ thống hiển thị trang workspace.
      \item Người dùng chọn xem danh sách thành viên trong workspace.
      \item Hệ thống hiển thị danh sách các thành viên trong workspace.
      \item Người dùng chọn nút 3 chấm bên cạnh tên thành viên cụ thể.
      \item Hệ thống hiển thị 1 dropdown menu.
      \item Người dùng chọn tùy chọn “Assign owner permission”
      \item Hệ thống hiển thị cảnh báo với nội dung: “Do you want to assign owner permission to this member? Your new permission will be a shared member who can edit workspace.
      \item Người dùng chọn “Agree”.
      \item Hệ thống chuyển quyền thành công và điều hướng người dùng tới trang workspace hiển thị cho shared member có thể edit. 
  \end{enumerate}
  \\ \hline
\textbf{Alternative Flow} & Không                                                                                                                   \\ \hline
\textbf{Exceptions}       & \begin{tabular}[c]{@{}l@{}}9.1. Người dùng chọn Cancel.\\ 10.1. Hệ thống tắt cảnh báo.\\ Usecase dừng lại.\end{tabular} \\ \hline
\end{tabular}%
}
\end{table}
